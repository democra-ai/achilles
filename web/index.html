Quick Start
# Fetch secrets for AI agent
curl -X POST http://localhost:8900/api/v1/ai/secrets \
  -H "X-API-Key: av_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"project": "my-app", "environment": "production", "keys": ["OPENAI_API_KEY"]}'

# MCP tool call
curl -X POST http://localhost:8900/api/v1/ai/mcp/call \
  -H "X-API-Key: av_your_key_here" \
  -H "Content-Type: application/json" \
  -d '{"name": "get_secret", "arguments": {"project": "my-app", "key": "DATABASE_URL"}}'
